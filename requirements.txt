# ============================================================
# Vision Assistant — Python Dependencies
# ============================================================
# Install with:
#   Windows  →  py -3.11 -m pip install -r requirements.txt
#   macOS    →  python3.11 -m pip install -r requirements.txt
#   Linux    →  python3.11 -m pip install -r requirements.txt
#
# NOTE: Tesseract OCR and Ollama must be installed separately.
# See CONFIG.txt for instructions.
# ============================================================

# ── GUI ─────────────────────────────────────────────────────
PyQt6                  # floating window, eye widget, overlay, text input bar

# ── Screen capture ──────────────────────────────────────────
mss                    # fast screen capture (core/capture.py)
Pillow                 # image processing, required by pytesseract

# ── OCR ─────────────────────────────────────────────────────
pytesseract            # Python wrapper for Tesseract OCR (core/icon_finder.py)

# ── Voice / STT ─────────────────────────────────────────────
RealtimeSTT            # always-on wake word + Whisper STT (voice/listener.py)
silero-vad             # voice activity detection — prevents ~45s startup delay

# ── TTS ─────────────────────────────────────────────────────
pyttsx3                # offline TTS — SAPI5 (Windows), nsss (macOS), espeak (Linux)

# ── Input automation ────────────────────────────────────────
pyautogui              # cross-platform mouse/keyboard automation

# ── Networking ──────────────────────────────────────────────
requests               # HTTP calls to local Ollama API (core/ollama_client.py)

# ── Windows-only (do NOT install on Linux or macOS) ─────────
# Uncomment these only if you are on Windows:
#
# pywinauto              # UI element detection (core/icon_finder.py) — Windows only
# pywin32                # Windows COM/API and win32gui for WhatsApp focus
# pygetwindow            # WhatsApp window polling (actions/dispatcher.py) — Windows only
#
# Linux users: install wmctrl or xdotool instead for WhatsApp focus:
#   sudo apt install wmctrl
# macOS users: no extra packages needed (uses osascript)
